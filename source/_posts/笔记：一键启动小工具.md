## 1. Py 代码

创建 `一键启动.py` 并编写代码：
```
import sys
import json
import subprocess
from PyQt5.QtWidgets import (
    QApplication, QMainWindow, QTableWidget, QTableWidgetItem, QCheckBox, 
    QPushButton, QDialog, QVBoxLayout, QHBoxLayout, QLabel, QLineEdit, 
    QFileDialog, QWidget
)
from PyQt5.QtCore import Qt

class AddExeDialog(QDialog):
    def __init__(self, parent=None):
        super(AddExeDialog, self).__init__(parent)
        self.setWindowTitle("添加EXE文件")
        self.setFixedSize(400, 200)

        # 居中显示对话框
        if parent:
            geo = parent.geometry()
            x = geo.x() + (geo.width() - self.width()) // 2
            y = geo.y() + (geo.height() - self.height()) // 2
            self.move(x, y)

        self.name_label = QLabel("名称：")
        self.name_input = QLineEdit()
        self.path_label = QLabel("路径：")
        self.path_input = QLineEdit()
        self.path_button = QPushButton("选择文件")
        self.path_button.clicked.connect(self.select_file)
        self.ok_button = QPushButton("确认")
        self.ok_button.clicked.connect(self.accept)
        self.cancel_button = QPushButton("取消")
        self.cancel_button.clicked.connect(self.reject)

        layout = QVBoxLayout()
        layout.addWidget(self.name_label)
        layout.addWidget(self.name_input)
        layout.addWidget(self.path_label)
        layout.addWidget(self.path_input)
        layout.addWidget(self.path_button)
        
        button_layout = QHBoxLayout()
        button_layout.addWidget(self.ok_button)
        button_layout.addWidget(self.cancel_button)
        layout.addLayout(button_layout)
        
        self.setLayout(layout)

    def select_file(self):
        file_path, _ = QFileDialog.getOpenFileName(self, "选择EXE文件", "", "EXE Files (*.exe)")
        if file_path:
            self.path_input.setText(file_path)

    def get_data(self):
        return self.name_input.text(), self.path_input.text()

class MainWindow(QMainWindow):
    def __init__(self):
        super(MainWindow, self).__init__()
        self.setWindowTitle("EXE文件启动器")
        self.setGeometry(100, 100, 700, 500)

        # Table setup
        self.table = QTableWidget()
        self.table.setColumnCount(3)
        self.table.setHorizontalHeaderLabels(["勾选", "名称", "路径"])
        self.table.setColumnWidth(0, 50)
        self.table.setColumnWidth(1, 150)
        self.table.setColumnWidth(2, 450)
        self.table.horizontalHeader().setStretchLastSection(True)

        # Buttons
        self.add_button = QPushButton("添加")
        self.add_button.setShortcut("Return")  # Enter key shortcut
        self.add_button.clicked.connect(self.add_exe)
        self.delete_button = QPushButton("删除")
        self.delete_button.clicked.connect(self.delete_exe)
        self.save_button = QPushButton("保存")
        self.save_button.clicked.connect(self.save_config)
        self.start_button = QPushButton("启动")
        self.start_button.clicked.connect(self.start_exes)

        # Button layout
        button_layout = QHBoxLayout()
        button_layout.addWidget(self.add_button)
        button_layout.addWidget(self.delete_button)
        button_layout.addWidget(self.save_button)
        button_layout.addWidget(self.start_button)
        button_layout.addStretch()

        # Main layout
        main_layout = QVBoxLayout()
        main_layout.addWidget(self.table)
        main_layout.addLayout(button_layout)

        central_widget = QWidget()
        central_widget.setLayout(main_layout)
        self.setCentralWidget(central_widget)

        # Load saved config
        self.load_config()

    def add_exe(self):
        dialog = AddExeDialog(self)
        if dialog.exec_():
            name, path = dialog.get_data()
            if name and path:
                row = self.table.rowCount()
                self.table.insertRow(row)
                check_box = QCheckBox()
                check_box.setChecked(True)  # Default checked
                self.table.setCellWidget(row, 0, check_box)
                self.table.setItem(row, 1, QTableWidgetItem(name))
                self.table.setItem(row, 2, QTableWidgetItem(path))
                self.save_config()  # Auto-save after adding

    def delete_exe(self):
        selected_rows = sorted(set(index.row() for index in self.table.selectedIndexes()), reverse=True)
        for row in selected_rows:
            self.table.removeRow(row)
        self.save_config()  # Auto-save after deleting

    def save_config(self):
        config = []
        for row in range(self.table.rowCount()):
            check_box = self.table.cellWidget(row, 0)
            name = self.table.item(row, 1).text()
            path = self.table.item(row, 2).text()
            config.append({"name": name, "path": path, "checked": check_box.isChecked()})
        try:
            with open("config.json", "w", encoding="utf-8") as f:
                json.dump(config, f, ensure_ascii=False, indent=4)
        except Exception as e:
            print(f"Failed to save config: {e}")

    def load_config(self):
        try:
            with open("config.json", "r", encoding="utf-8") as f:
                config = json.load(f)
                self.table.setRowCount(0)  # Clear existing rows
                for item in config:
                    row = self.table.rowCount()
                    self.table.insertRow(row)
                    check_box = QCheckBox()
                    check_box.setChecked(item.get("checked", True))  # Use get with default
                    self.table.setCellWidget(row, 0, check_box)
                    self.table.setItem(row, 1, QTableWidgetItem(item["name"]))
                    self.table.setItem(row, 2, QTableWidgetItem(item["path"]))
        except FileNotFoundError:
            pass  # No config file exists yet
        except Exception as e:
            print(f"Failed to load config: {e}")

    def start_exes(self):
        for row in range(self.table.rowCount()):
            check_box = self.table.cellWidget(row, 0)
            if check_box.isChecked():
                path = self.table.item(row, 2).text()
                try:
                    subprocess.Popen(path)
                except Exception as e:
                    print(f"Failed to start {path}: {e}")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    
    # Apply QSS for modern UI styling
    qss = """
    QMainWindow {
        background-color: #f0f2f5;
    }
    QTableWidget {
        background-color: #ffffff;
        border: 1px solid #dcdcdc;
        border-radius: 5px;
        font-size: 14px;
    }
    QTableWidget::item {
        padding: 5px;
    }
    QHeaderView::section {
        background-color: #e8ecef;
        padding: 5px;
        border: 1px solid #dcdcdc;
        font-weight: bold;
    }
    QPushButton {
        background-color: #007bff;
        color: white;
        border: none;
        padding: 8px 15px;
        border-radius: 5px;
        font-size: 14px;
    }
    QPushButton:hover {
        background-color: #0056b3;
    }
    QLineEdit {
        border: 1px solid #dcdcdc;
        border-radius: 5px;
        padding: 5px;
        font-size: 14px;
    }
    QLabel {
        font-size: 14px;
    }
    QDialog {
        background-color: #ffffff;
        border: 1px solid #dcdcdc;
        border-radius: 5px;
    }
    """
    app.setStyleSheet(qss)
    
    window = MainWindow()
    window.show()
    sys.exit(app.exec_())
```
